SeedJs

Use your javascript console to play with the provided examples of classes: 

        Vehicle = Seed.extend({
            engine:{cilindros:0, pistones:0},
            wheels:0,
            doors:0,
            speed: 0,
            noise: "rrrrrrrr",
            run: function() {
                $('body').append(this.noise);
            },
            accelerate: function() {
                this.speed += 10;
                console.log('new speed:' + this.speed);
            },
            brake: function() {
                this.speed -= 10;
            }
        });
        
        Car = Vehicle.extend({wheels: 4,

        Motorbike = Vehicle.extend({wheels: 2});
        Clio = Car.extend({
            engine:{cilindros:8, pistones:16},
            accelerate: function() {
                this.parent('accelerate');
                if(this.speed > 50) {
                    this.speed = 0;
                    alert('CRASH');
                }
            },
            run: function() {
                this.accelerate();
                this.parent('run');
            }
