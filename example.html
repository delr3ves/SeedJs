SeedJs

Use your javascript console to play with the provided examples of classes: 

        Vehicle = Seed.extend({
            motor:{cilindros:0, pistones:0},
            ruedas:0,
            puertas:0,
            speed: 0,
            sound: "rrrrrrrr",
            run: function() {
                $('body').append(this.sound);
            },
            accelerate: function() {
                this.speed += 10;
                console.log('new speed:' + this.speed);
            },
            brake: function() {
                this.speed -= 10;
            }
        });
        
        Car = Vehicle.extend({ruedas: 4,

            sound: "brrum brrrum"
        });
        Moto = Vehicle.extend({ruedas: 2});
        Clio = Car.extend({
            motor:{cilindros:8, pistones:16},
            accelerate: function() {
                this.parent('accelerate');
                if(this.speed > 50) {
                    this.speed = 0;
                    alert('CRASH');
                }
            },
            run: function() {
                this.accelerate();
                this.parent('run');
            }

        });
